ALTER proc [dbo].[sp_admin_attach_routine](
	@schid nvarchar(50) = null,
	@flag nchar(2) = null,
	@userid int = null,
	@classid int = null,
	@sectionid int = null,
	@termid int = null,
	@routineid int = null,
	@routtype nvarchar(4) = null,
	@attachment nvarchar(250) = null,
	@imgurl nvarchar(250) = null
)
AS
BEGIN
	If(@flag='i')
	Begin
		
		If not exists(Select 'm' from tbl_user where sch_id=@schid and user_id=@userid and del_flag='N')
		Begin
			Select 203 StatusCode, 'Student does not exists' Message
			return;
		End

		If  exists(Select 'm' from tbl_attach_routine where sch_id=@schid and class_id=@classid and section_id=@sectionid and @routtype='C' and is_del='N') and @routtype='C'
		Begin
			Select 203 StatusCode, 'Class Routine already exist' Message
			return;
		End

		If  exists(Select 'm' from tbl_attach_routine where sch_id=@schid and class_id=@classid and term_id=@termid and @routtype='E' and is_del='N') and @routtype='C'
		Begin
			Select 203 StatusCode, 'Class Routine already exist' Message
			return;
		End

		insert into tbl_attach_routine(sch_id,user_id,class_id,section_id,term_id,rout_type,attach,created_date,updated_date,is_del) 
		values (@schid,@userid,@classid,@sectionid,@termid,@routtype,@attachment,GETDATE(),GETDATE(),'N');		
			Select 200 StatusCode, 'Success.' Message;
		return;
	End

	If(@flag='S')
	Begin

	 if(@routtype='C')
	 BEGIN
	select ID RoutineID,'One' Class,'A' Section,@imgurl+'document/'+attach Attachment from tbl_attach_routine where rout_type='C' and sch_id=@schid and is_del='N'
	return;
	END;

	 if(@routtype='E')
	 BEGIN
	select ID RoutineID,'One' Class,'First Terminal Exam' Term,@imgurl+'document/'+attach Attachment from tbl_attach_routine where rout_type='E' and sch_id=@schid and is_del='N'
	return;
	END;

	if(@routtype='B')
	BEGIN	
	select exam_name Title,'Exam' Type,@imgurl+'document/'+attach Attachment from tbl_attach_routine r
	left join tbl_term t on  t.term_id=r.term_id and t.sch_id=r.sch_id
	where r.class_id=@classid  and r.rout_type='E' and r.sch_id=@schid and r.is_del='N'
	UNION
	select 'Class Routine' Title,'Class' Type,@imgurl+'document/'+attach Attachment from tbl_attach_routine where 
	class_id=@classid and section_id=@sectionid and rout_type='C' and sch_id=@schid and is_del='N'
	return;
	END;


	END
	If(@flag='R')
	Begin
		
		If not exists(Select 'm' from tbl_user where sch_id=@schid and user_id=@userid and del_flag='N')
		Begin
			Select 203 StatusCode, 'Student does not exists' Message
			return;
		End

		If not exists(Select 'm' from tbl_attach_routine where sch_id=@schid and id=@routineid and is_del='N')
		Begin
			Select 203 StatusCode, 'Routine doesnot exist.' Message
			return;
		End
	
	update tbl_attach_routine set is_del='Y',updated_date=GETDATE() where sch_id=@schid and id=@routineid;
	Select 200 StatusCode, 'Success.' Message;
		return;
	End

	
END
