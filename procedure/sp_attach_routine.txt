ALTER proc [dbo].[sp_attach_routine](
	@schid nvarchar(50) = null,
	@flag nchar(2) = null,
	@userid int = null,
	@classid int = null,
	@sectionid int = null,
	@termid int = null,
	@routineid int = null,
	@routtype nvarchar(4) = null,
	@attachment nvarchar(250) = null,
	@imgurl nvarchar(250) = null
)
AS
BEGIN

	
	If(@flag='S')
	Begin

	 if(@routtype='C')
	 BEGIN
	select rt.ID RoutineID,clName Class,secName Section,@imgurl+'document/'+attach Attachment 
	from tbl_attach_routine rt
	left join stClass c on c.id=rt.class_id
	left join stSection s on s.id=rt.section_id
	where 
	 class_id=@classid and section_id=@sectionid and rout_type='C' and sch_id=@schid and is_del='N'
	return;
	END;

	 if(@routtype='E')
	 BEGIN
	select rt.ID RoutineID,clName Class,examTerminal Term,@imgurl+'document/'+attach Attachment from tbl_attach_routine 
	left join stClass c on c.id=rt.class_id
	left join msheetTerminal t on t.id=rt.term_id
	where class_id=@classid and term_id=@termid and rout_type='E' and sch_id=@schid and is_del='N'
	return;
	END;

	END

	
END
